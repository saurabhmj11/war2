    83→export default function MindMirrorPage() {
    84→  // ===== STATE =====
    85→  const [messages, setMessages] = useState<Message[]>([
    86→    {
    87→      id: 'welcome',
    88→      role: 'ai',
    89→      text: "Hi! I'm **MindMirror** — your personal wellness companion. I'm here to listen without judgment, and sometimes I might notice things in your words that even you haven't fully realized yet.\n\nHow are you feeling about your exam preparation today?",
    90→      time: 'Just now'
    91→    }
    92→  ]);
    93→  const [inputText, setInputText] = useState('');
    94→  const [isLoading, setIsLoading] = useState(false);
    95→  const [activeView, setActiveView] = useState<'chat' | 'journal' | 'privacy'>('chat');
    96→  const [showMobilePanel, setShowMobilePanel] = useState(false);
    97→
    98→  // Real sensor analysis (camera + microphone)
    99→  const sensor = useSensorAnalysis();
   100→
   101→  // Insight panel state
   102→  const [textSentiment, setTextSentiment] = useState('neutral');
   103→  const [triggers, setTriggers] = useState<Record<string, number>>({});
   104→  const [burnout, setBurnout] = useState(0);
   105→  const [moodHistory, setMoodHistory] = useState<{ label: string; score: number }[]>([]);
   106→  const [insight, setInsight] = useState('');
   107→  const [showInsight, setShowInsight] = useState(false);
   108→  const [contradiction, setContradiction] = useState<{ detected: boolean; type: string; question: string } | null>(null);
   109→  const [exchangeCount, setExchangeCount] = useState(0);
   110→
   111→  // Use real sensor values
   112→  const voiceTone = sensor.voiceTone;
   113→  const faceSignal = sensor.faceSignal;
   114→
   115→  // Journal state
   116→  const [journalEntries, setJournalEntries] = useState<JournalEntry[]>([]);
   117→
   118→  // Settings state
   119→  const [settings, setSettings] = useState({ camera: true, voice: true, journal: true, crisis: true });
   120→
   121→  // Refs
   122→  const conversationHistory = useRef<{ role: string; content: string }[]>([]);
   123→  const messagesEndRef = useRef<HTMLDivElement>(null);
   124→  const textareaRef = useRef<HTMLTextAreaElement>(null);
   125→
   126→  // ===== LOAD SETTINGS & JOURNAL =====
   127→  useEffect(() => {
   128→    try {
   129→      const savedSettings = { camera: true, voice: true, journal: true, crisis: true };
   130→      ['camera', 'voice', 'journal', 'crisis'].forEach(key => {
   131→        const stored = localStorage.getItem('mm_setting_' + key);
   132→        if (stored !== null) (savedSettings as Record<string, boolean>)[key] = stored === 'true';
   133→      });
   134→      setSettings(savedSettings);
   135→
   136→      const entries = JSON.parse(localStorage.getItem('mindmirror_journal') || '[]') as JournalEntry[];
   137→      setJournalEntries(entries);
   138→    } catch { /* ignore */ }
   139→  }, []);
   140→
   141→  // ===== AUTO-SCROLL =====
   142→  useEffect(() => {
   143→    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
   144→  }, [messages]);
   145→
   146→  // ===== AUTO-RESIZE TEXTAREA =====
   147→  useEffect(() => {
   148→    if (textareaRef.current) {
   149→      textareaRef.current.style.height = 'auto';
   150→      textareaRef.current.style.height = Math.min(textareaRef.current.scrollHeight, 120) + 'px';
   151→    }
   152→  }, [inputText]);
   153→
   154→  // ===== HELPERS =====
   155→  const getTime = () => new Date().toLocaleTimeString('en-IN', { hour: '2-digit', minute: '2-digit' });
   156→
   157→  const extractTags = (text: string): string[] => {
   158→    const subjects = ['Physics', 'Chemistry', 'Math', 'Maths', 'Biology', 'English', 'History', 'Geography', 'Economics', 'UPSC', 'JEE', 'NEET', 'CAT', 'GATE'];
   159→    const tags: string[] = [];
   160→    subjects.forEach(s => { if (text.toLowerCase().includes(s.toLowerCase())) tags.push(s); });
   161→    if (/stress|stressed|pressure/i.test(text)) tags.push('stress');
   162→    if (/tired|exhaust|burnout/i.test(text)) tags.push('burnout');
   163→    if (/can't|cannot|give up/i.test(text)) tags.push('self-doubt');
   164→    if (/anxious|worry|nervous/i.test(text)) tags.push('anxiety');
   165→    return tags.slice(0, 4);
   166→  };
   167→
   168→  // ===== SAVE TO JOURNAL =====
   169→  const saveToJournal = useCallback((currentMessages: Message[]) => {
   170→    try {
   171→      const summary = currentMessages
   172→        .filter(m => m.role === 'user')
   173→        .map(m => m.text)
   174→        .join(' ');
   175→      if (!summary.trim()) return;
   176→
   177→      const entry: JournalEntry = {
   178→        id: Date.now(),
   179→        date: new Date().toLocaleString('en-IN'),
   180→        summary: summary.slice(0, 300),
   181→        tags: extractTags(summary),
   182→        mood: textSentiment,
   183→      };
   184→      const entries = [entry, ...journalEntries].slice(0, 50);
   185→      setJournalEntries(entries);
   186→      localStorage.setItem('mindmirror_journal', JSON.stringify(entries));
   187→    } catch { /* ignore */ }
   188→  }, [textSentiment, journalEntries]);
   189→
   190→  // ===== FORMAT TEXT WITH MARKDOWN-LIKE BOLD =====
   191→  const formatText = (text: string) => {
   192→    return text
   193→      .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
   194→      .replace(/\n/g, '<br>');
   195→  };
   196→
   197→  // ===== SEND MESSAGE =====
   198→  const sendMessage = async (overrideText?: string) => {
   199→    const text = (overrideText || inputText).trim();
   200→    if (!text || isLoading) return;
   201→
   202→    setIsLoading(true);
   203→    setInputText('');
   204→
   205→    // Reset textarea height
   206→    if (textareaRef.current) {
   207→      textareaRef.current.style.height = 'auto';
   208→    }
   209→
   210→    // ===== CLIENT-SIDE CRISIS DETECTION =====
   211→    const CRISIS_KEYWORDS = ['give up on everything', 'end it all', 'no point living', 'suicide', 'kill myself', 'don\'t want to live', 'want to die', 'end my life', 'can\'t go on', 'better off dead'];
   212→    const isClientCrisis = CRISIS_KEYWORDS.some(kw => text.toLowerCase().includes(kw));
   213→
   214→    // ===== CLIENT-SIDE TRIGGER DETECTION =====
   215→    const subjectKeywords: Record<string, string[]> = {
   216→      'Physics': ['physics', 'mechanics', 'thermodynamics', 'electro'],
   217→      'Chemistry': ['chemistry', 'organic', 'inorganic', 'reactions'],
   218→      'Math': ['math', 'maths', 'calculus', 'algebra', 'trigonometry'],
   219→      'Biology': ['biology', 'botany', 'zoology'],
   220→      'Parents': ['parents', 'mom', 'dad', 'family pressure', 'family expects'],
   221→      'Sleep': ['sleep', 'insomnia', 'can\'t sleep', 'tired', 'exhausted'],
   222→      'Time': ['time', 'running out of time', 'deadline', 'not enough time'],
   223→      'Mock Tests': ['mock test', 'mock exam', 'practice test', 'test score'],
   224→    };
   225→    const clientTriggers: Record<string, number> = {};
   226→    for (const [subject, keywords] of Object.entries(subjectKeywords)) {
   227→      if (keywords.some(kw => text.toLowerCase().includes(kw))) {
   228→        clientTriggers[subject] = 65;
   229→      }
   230→    }
   231→
   232→    // ===== CLIENT-SIDE SENTIMENT HINT =====
   233→    const negativeWords = ['stressed', 'anxious', 'worried', 'scared', 'overwhelmed', 'can\'t', 'hate', 'terrible', 'awful', 'impossible', 'hopeless', 'give up', 'burnout', 'exhausted'];
   234→    const isNegativeText = negativeWords.some(w => text.toLowerCase().includes(w));
   235→
   236→    const userMsg: Message = { id: `u-${Date.now()}`, role: 'user', text, time: getTime() };
   237→    const newMessages = [...messages, userMsg];
   238→    setMessages(newMessages);
   239→
   240→    // Add thinking indicator
   241→    const thinkingMsg: Message = { id: 'thinking', role: 'ai', text: '', time: '' };
   242→    setMessages([...newMessages, thinkingMsg]);
   243→
   244→    conversationHistory.current.push({ role: 'user', content: text });
   245→
   246→    // Immediately apply client-side triggers to panel
   247→    if (Object.keys(clientTriggers).length > 0) {
   248→      setTriggers(prev => {
   249→        const merged = { ...prev };
   250→        for (const [key, val] of Object.entries(clientTriggers)) {
   251→          merged[key] = Math.max(merged[key] || 0, val);
   252→        }
   253→        const sorted = Object.entries(merged).sort(([, a], [, b]) => b - a).slice(0, 8);
   254→        return Object.fromEntries(sorted);
   255→      });
   256→    }
   257→
   258→    // Immediately increment burnout slightly for negative messages
   259→    if (isNegativeText) {
   260→      setBurnout(prev => Math.min(100, prev + 8));
   261→    }
   262→
   263→    try {
   264→      const response = await fetch('/api/chat', {
   265→        method: 'POST',
   266→        headers: { 'Content-Type': 'application/json' },
   267→        body: JSON.stringify({
   268→          messages: conversationHistory.current.slice(-20),
   269→          faceSignal: faceSignal !== 'relaxed' ? faceSignal : undefined,
   270→          voiceTone: voiceTone !== 'calm' ? voiceTone : undefined,
   271→        }),
   272→      });
   273→
   274→      if (!response.ok) throw new Error('API error');
   275→
   276→      const data = await response.json();
   277→      const { reply, meta } = data;
   278→
   279→      const aiMsg: Message = {
   280→        id: `a-${Date.now()}`,
   281→        role: 'ai',
   282→        text: reply || "I'm here. Could you tell me more?",
   283→        time: getTime(),
   284→        meta
   285→      };
   286→      const messagesAfterAI = [...newMessages, aiMsg];
   287→
   288→      // Add crisis message if flagged by LLM OR client-side detection
   289→      if ((meta?.crisisFlag || isClientCrisis) && settings.crisis) {
   290→        const crisisMsg: Message = {
   291→          id: `c-${Date.now()}`,
   292→          role: 'ai',
   293→          text: "I want to make sure you're okay. If things feel really heavy right now, please reach out:\n\n📞 **iCall**: 9152987821\n📞 **Vandrevala Foundation**: 1860-2662-345\n\nThey're trained to listen, and you deserve support beyond what I can offer.",
   294→          time: getTime(),
   295→          isCrisis: true
   296→        };
   297→        messagesAfterAI.push(crisisMsg);
   298→      }
   299→
   300→      setMessages(messagesAfterAI);
   301→      conversationHistory.current.push({ role: 'assistant', content: reply });
   302→
   303→      // Update exchange count
   304→      setExchangeCount(prev => prev + 1);
   305→
   306→      // Update insight panel with LLM meta + client-side enhancements
   307→      if (meta) {
   308→        if (meta.textSentiment) setTextSentiment(meta.textSentiment);
   309→        // Note: voiceTone and faceSignal now come from real sensors (useSensorAnalysis)
   310→        // We still log LLM's interpretation for debugging but don't override real sensor data
   311→        
   312→        // Merge triggers from LLM with client-side triggers
   313→        if (meta.triggers && Object.keys(meta.triggers).length > 0) {
   314→          setTriggers(prev => {
   315→            const merged = { ...prev, ...meta.triggers };
   316→            const sorted = Object.entries(merged).sort(([, a], [, b]) => b - a).slice(0, 8);
   317→            return Object.fromEntries(sorted);
   318→          });
   319→        }
   320→        
   321→        // Use LLM burnout if higher than current, or increment client-side
   322→        if (typeof meta.burnout === 'number') {
   323→          setBurnout(prev => Math.max(prev, meta.burnout));
   324→        }
   325→        
   326→        // Add mood score to history
   327→        if (meta.textSentiment) {
   328→          const score = moodScoreMap[meta.textSentiment] ?? 50;
   329→          setMoodHistory(prev => {
   330→            const next = [...prev, { label: `#${prev.length + 1}`, score }].slice(-10);
   331→            return next;
   332→          });
   333→        }
   334→        
   335→        // Show insight from LLM or generate client-side insight
   336→        if (meta.insight) {
   337→          setInsight(meta.insight);
   338→          setShowInsight(true);
   339→        } else if (exchangeCount >= 2) {
   340→          // Client-side insight generation as fallback
   341→          const triggerNames = Object.keys(triggers);
   342→          if (triggerNames.length > 0) {
   343→            const topTrigger = triggerNames[0];
   344→            const negWords = ['stressed', 'anxious', 'worried', 'overwhelmed', 'can\'t'];
   345→            const hasNeg = conversationHistory.current
   346→              .filter(m => m.role === 'user')
   347→              .some(m => negWords.some(w => m.content.toLowerCase().includes(w)));
   348→            if (hasNeg) {
   349→              setInsight(`${topTrigger} appears frequently alongside your stressful thoughts — this might be a knowledge-gap anxiety pattern, not a capability issue.`);
   350→              setShowInsight(true);
   351→            }
   352→          }
   353→        }
   354→        
   355→        if (meta.contradictionDetected) {
   356→          setContradiction({
   357→            detected: true,
   358→            type: meta.contradictionType || 'verbal-vs-facial',
   359→            question: meta.adaptiveQuestion || ''
   360→          });
   361→        } else if (isNegativeText && text.toLowerCase().includes('fine')) {
   362→          // Client-side contradiction: "fine" + negative context
   363→          setContradiction({
   364→            detected: true,
   365→            type: 'verbal-vs-behavioral',
   366→            question: 'You say you\'re fine, but your words suggest otherwise. What\'s really going on?'
   367→          });
   368→        } else {
   369→          setContradiction(null);
   370→        }
   371→      } else {
   372→        // No meta from LLM — still apply client-side updates
   373→        if (isNegativeText) {
   374→          setMoodHistory(prev => {
   375→            const next = [...prev, { label: `#${prev.length + 1}`, score: 30 }].slice(-10);
   376→            return next;
   377→          });
   378→        }
   379→      }
   380→
   381→      // Save to journal
   382→      if (settings.journal) {
   383→        saveToJournal(messagesAfterAI);
   384→      }
   385→    } catch (err) {

   387→        id: `e-${Date.now()}`, role: 'ai', text: "I had trouble connecting. Please try again.", time: getTime()
   388→      }));
   389→      console.error('API error:', err);
   390→    }
   391→
   392→    setIsLoading(false);
   393→  };
   394→
   395→  // ===== CLEAR SESSION =====
   396→  const clearSession = () => {
   397→    if (!confirm('Start a new session? Current conversation will be saved to the journal.')) return;
   398→    if (settings.journal) saveToJournal(messages);
   399→    conversationHistory.current = [];
   400→    setMessages([
   401→      { id: 'new-welcome', role: 'ai', text: "Hi again! Ready for a new session. How are you feeling right now?", time: 'Just now' }
   402→    ]);
   403→    setMoodHistory([]);
   404→    setBurnout(0);
   405→    setTriggers({});
   406→    setInsight('');
   407→    setShowInsight(false);
   408→    setContradiction(null);
   409→    setTextSentiment('neutral');
   410→    // Note: voiceTone and faceSignal reset automatically when camera/mic stop
   411→    setExchangeCount(0);
   412→  };
   413→
   414→  // ===== EXPORT / DELETE DATA =====
   415→  const exportData = () => {
   416→    try {
   417→      const blob = new Blob([JSON.stringify(journalEntries, null, 2)], { type: 'application/json' });
   418→      const url = URL.createObjectURL(blob);
   419→      const a = document.createElement('a');
   420→      a.href = url; a.download = 'mindmirror_journal.json'; a.click();
   421→      URL.revokeObjectURL(url);
   422→    } catch { /* ignore */ }
   423→  };
   424→
   425→  const deleteData = () => {
   426→    if (!confirm('Delete all journal entries? This cannot be undone.')) return;
   427→    localStorage.removeItem('mindmirror_journal');
   428→    setJournalEntries([]);
   429→  };
   430→
   431→  // ===== DERIVED STATE =====
   432→  const sortedTriggers = Object.entries(triggers).sort(([, a], [, b]) => b - a).slice(0, 6);
   433→  const burnoutLabel = burnout >= 70 ? 'High' : burnout >= 40 ? 'Moderate' : burnout > 0 ? 'Low' : '—';
   434→  const burnoutColor = burnout >= 70 ? 'text-rose-600' : burnout >= 40 ? 'text-amber-600' : 'text-emerald-600';
   435→  const burnoutBarColor = burnout >= 70 ? 'bg-rose-500' : burnout >= 40 ? 'bg-amber-500' : 'bg-emerald-500';
   436→  const burnoutDesc = burnout >= 70 ? 'Consider taking a real break today.' : burnout >= 40 ? "You're pushing hard. Rest matters." : burnout > 0 ? "You're managing well right now." : 'No data yet';
   437→
   438→  // ===== SVG CHART FOR MOOD TIMELINE =====
   439→  const renderMoodChart = () => {
   440→    if (moodHistory.length === 0) {
   441→      return (
   442→        <div className="text-xs text-slate-400 text-center py-4">
   443→          Start talking — mood trend will appear
   444→        </div>
   445→      );
   446→    }
   447→
   448→    const w = 280;
   449→    const h = 80;
   450→    const padX = 24;
   451→    const padY = 10;
   452→    const chartW = w - padX * 2;
   453→    const chartH = h - padY * 2;
   454→
   455→    // Handle single-point case (show just a dot)
   456→    if (moodHistory.length === 1) {
   457→      const m = moodHistory[0];
   458→      const px = padX + chartW / 2;
   459→      const py = padY + chartH - (m.score / 100) * chartH;
   460→      const moodLabel = Object.entries(moodScoreMap).find(([, v]) => v === m.score)?.[0] || '';
   461→
   462→      return (
   463→        <svg viewBox={`0 0 ${w} ${h}`} className="w-full" aria-label="Mood trend chart">
   464→          {[0, 25, 50, 75, 100].map(v => {
   465→            const y = padY + chartH - (v / 100) * chartH;
   466→            return <line key={v} x1={padX} y1={y} x2={w - padX} y2={y} stroke="#e2e8f0" strokeWidth="0.5" strokeDasharray="3 3" />;

   468→          <circle cx={px} cy={py} r="5" fill="#7c3aed" stroke="#fff" strokeWidth="2" />
   469→          <text x={w / 2} y={h - 2} textAnchor="middle" className="text-[8px] fill-slate-400" fontFamily="Inter, sans-serif">{moodLabel}</text>
   470→        </svg>
   471→      );
   472→    }
   473→
   474→    const points = moodHistory.map((m, i) => ({
   475→      x: padX + (i / Math.max(moodHistory.length - 1, 1)) * chartW,
   476→      y: padY + chartH - (m.score / 100) * chartH,
   477→      score: m.score,
   478→      label: m.label
   479→    }));
   480→
   481→    const linePath = points.map((p, i) => `${i === 0 ? 'M' : 'L'} ${p.x} ${p.y}`).join(' ');
   482→    const areaPath = `${linePath} L ${points[points.length - 1].x} ${padY + chartH} L ${points[0].x} ${padY + chartH} Z`;
   483→
   484→    return (
   485→      <svg viewBox={`0 0 ${w} ${h}`} className="w-full" aria-label="Mood trend chart">
   486→        {/* Grid lines */}
   487→        {[0, 25, 50, 75, 100].map(v => {
   488→          const y = padY + chartH - (v / 100) * chartH;
   489→          return <line key={v} x1={padX} y1={y} x2={w - padX} y2={y} stroke="#e2e8f0" strokeWidth="0.5" strokeDasharray="3 3" />;
   490→        })}
   491→        {/* Area fill */}
   492→        <path d={areaPath} fill="url(#moodGrad)" opacity="0.3" />
   493→        {/* Line */}
